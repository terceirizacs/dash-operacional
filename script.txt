    { dadosAba: "Dra Investidora", kpiAba: "Operação Semanal Dra Investidor", meta: 50 },

    { dadosAba: "Lucas Arrial", kpiAba: "Operação Semanal Lucas Arrial", meta: 500 },
    { dadosAba: "Elisa Fingermann", kpiAba: "Operação Semanal Elisa Fingerman", meta: 200 }
  ];

  var resultado = { clientes: {}, atualizadoEm: new Date().toLocaleString("pt-BR") };

  for (var i = 0; i < clientesConfig.length; i++) {
    var cfg = clientesConfig[i];
    var nome = cfg.dadosAba;

    var dados = processarDados(ss, cfg.dadosAba);

      meta: cfg.meta,
      porDia: dados.porDia,
      totais: dados.totais,
      sla: dados.sla
    };
  }

  return ContentService.createTextOutput(JSON.stringify(resultado))
    .setMimeType(ContentService.MimeType.JSON);
}

function extrairChaveData(val) {
  if (!val || val === '' || val === 'undefined') return null;
  // Se é Date object (Google Sheets)
  if (val instanceof Date && !isNaN(val.getTime())) {
    var d = val.getDate().toString().padStart(2, '0');
    var m = (val.getMonth() + 1).toString().padStart(2, '0');
    var y = val.getFullYear();
    return y + '-' + m + '-' + d;
  }
  // Se é string, tentar DD/MM/YYYY
  var s = String(val).trim();
  if (s === '' || s === 'undefined') return null;
  var match = s.match(/(\d{2})\/(\d{2})\/(\d{4})/);
  if (match) return match[3] + '-' + match[2] + '-' + match[1];
  return null;
}

function extrairTimestamp(val) {
  if (!val || val === '' || val === 'undefined') return null;
  if (val instanceof Date && !isNaN(val.getTime())) return val.getTime();
  var s = String(val).trim();
  if (s === '' || s === 'undefined') return null;
  var m = s.match(/(\d{2})\/(\d{2})\/(\d{4})\s+(\d{2}):(\d{2})/);
  if (m) return new Date(+m[3], +m[2]-1, +m[1], +m[4], +m[5]).getTime();
  var m2 = s.match(/(\d{2})\/(\d{2})\/(\d{4})/);
  if (m2) return new Date(+m2[3], +m2[2]-1, +m2[1]).getTime();
  return null;
}

function processarDados(ss, nomeAba) {
  var sheet = ss.getSheetByName(nomeAba);
  // Fallback: busca aba que contenha o nome (insensitive a espaços/case)
  if (!sheet) {
    var normalize = function(s){ return (s||'').toString().toLowerCase().replace(/\s+/g,' ').trim(); };
    var alvoNorm = normalize(nomeAba);
    var allSheets = ss.getSheets();
    for (var s = 0; s < allSheets.length; s++) {
      if (normalize(allSheets[s].getName()) === alvoNorm) {
        sheet = allSheets[s];
        break;
      }
    }
    // Fallback 2: busca por substring
    if (!sheet) {
      var partes = alvoNorm.split(' ').filter(Boolean);
      for (var ss2 = 0; ss2 < allSheets.length; ss2++) {
        var n = normalize(allSheets[ss2].getName());
        if (partes.every(function(p){ return n.indexOf(p) >= 0; })) {
          sheet = allSheets[ss2];
          break;
        }
      }
    }
  }
  var resultado = {
    porDia: {},
    totais: { enviados: 0, respostas: 0, respostasCS: 0, recontatos: 0, disparoUnicos: 0, comResposta: 0, semResposta: 0, totalLinhas: 0 },
    sla: { somaMinutosCS: 0, countCS: 0 }
  };

  if (!sheet) return resultado;

  var lastRow = sheet.getLastRow();
  if (lastRow < 2) return resultado;

  // Colunas: A = Status Disparo, B = Mensagem Agente, C = Mensagem Aluno
  var range = sheet.getRange(2, 1, lastRow - 1, 3);
  var values = range.getValues();

  for (var i = 0; i < values.length; i++) {
    var statusRaw = String(values[i][0]).trim();
    var alunoVal = values[i][2];
    var agenteVal = values[i][1];
    var alunoStr = String(alunoVal).trim();
    var agenteStr = String(agenteVal).trim();

    // Contar engajamento para TODAS as linhas (mesmo sem status)
    var temAluno = alunoStr !== '' && alunoStr !== 'undefined' && alunoVal !== null;
    var temAgente = agenteStr !== '' && agenteStr !== 'undefined' && agenteVal !== null;

    // Se tem status OU tem resposta do aluno, conta como linha válida
    var temStatus = statusRaw !== '' && statusRaw !== 'undefined';
    if (!temStatus && !temAluno && !temAgente) continue;

    // Contabilizar linha processada
    resultado.totais.totalLinhas++;

    // Engajamento: coluna C tem conteúdo?
    if (temAluno) {
      resultado.totais.comResposta++;
    } else {
      resultado.totais.semResposta++;
    }

    // Resposta aluno: extrair data
    var chaveAluno = extrairChaveData(alunoVal);
    if (chaveAluno) {
      if (!resultado.porDia[chaveAluno]) resultado.porDia[chaveAluno] = { enviados: 0, respostas: 0, respostasCS: 0 };
      resultado.porDia[chaveAluno].respostas++;
      resultado.totais.respostas++;
    }

    // Resposta agente: extrair data
    var chaveAgente = extrairChaveData(agenteVal);
    if (chaveAgente) {
      if (!resultado.porDia[chaveAgente]) resultado.porDia[chaveAgente] = { enviados: 0, respostas: 0, respostasCS: 0 };
      resultado.porDia[chaveAgente].respostasCS++;
      resultado.totais.respostasCS++;
    }

    // SLA CS: tempo entre resposta do aluno e resposta do agente
    if (temAluno && temAgente) {
      var tsAluno = extrairTimestamp(alunoVal);
      var tsAgente = extrairTimestamp(agenteVal);
      if (tsAluno && tsAgente && tsAgente > tsAluno) {
        var diffMin = Math.round((tsAgente - tsAluno) / 60000);
        if (diffMin >= 0 && diffMin < 14400) { // max 10 dias
          resultado.sla.somaMinutosCS += diffMin;
          resultado.sla.countCS++;
        }
      }
    }

    // Se não tem status de disparo, pular contagem de ✅
    if (!temStatus) continue;

    // Contar checks verdes na linha para follow-up/recontato
    var checksNaLinha = 0;

    // Extrair cada check verde com sua data
    var partes = statusRaw.split("+");
    for (var p = 0; p < partes.length; p++) {
      var parte = partes[p].trim();
      // Aceita ✅ (U+2705), ☑ (U+2611), ✔ (U+2714), ✓ (U+2713)
      if (parte.indexOf("✅") === -1 && parte.indexOf("☑") === -1 && parte.indexOf("✔") === -1 && parte.indexOf("✓") === -1) continue;

      checksNaLinha++;

      // Extrair data do formato: "✅ DD/MM/YYYY HH:MM"
      var matchData = parte.match(/(\d{2})\/(\d{2})\/(\d{4})/);
      if (!matchData) continue;

      var dia = matchData[1];
      var mes = matchData[2];
      var ano = matchData[3];
      var chave = ano + "-" + mes + "-" + dia; // formato YYYY-MM-DD

      if (!resultado.porDia[chave]) {
        resultado.porDia[chave] = { enviados: 0, respostas: 0, respostasCS: 0 };
      }

      resultado.porDia[chave].enviados++;
      resultado.totais.enviados++;
    }

    // Classificar recontato vs disparo unico
    if (checksNaLinha >= 2) {
      resultado.totais.recontatos++;
    } else if (checksNaLinha === 1) {
      resultado.totais.disparoUnicos++;
    }

    // (engajamento, respostas aluno e agente ja contabilizados no inicio do loop)
  }

  return resultado;
}

function listarAbas() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheets = ss.getSheets();
  Logger.log('=== ' + sheets.length + ' abas encontradas ===');
  for (var i = 0; i < sheets.length; i++) {
    var s = sheets[i];
    Logger.log((i+1) + '. "' + s.getName() + '" — ' + s.getLastRow() + ' linhas');
  }
}
