function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  var clientesConfig = [
    { dadosAba: "Andrea Vermont", kpiAba: "Operação Semanal Andrea", meta: 1000 },
    { dadosAba: "Dra Investidora", kpiAba: "Operação Semanal Dra Investidor", meta: 50 },
    { dadosAba: "Rafa Bueno", kpiAba: "Operação Semanal Rafa Bueno", meta: 250 },
    { dadosAba: "Bruno Lucarelli", kpiAba: "Operação Semanal Bruno Lucarell", meta: 200 }
  ];

  var resultado = { clientes: {}, atualizadoEm: new Date().toLocaleString("pt-BR") };

  for (var i = 0; i < clientesConfig.length; i++) {
    var cfg = clientesConfig[i];
    var nome = cfg.dadosAba;

    var dados = processarDados(ss, cfg.dadosAba);

    resultado.clientes[nome] = {
      meta: cfg.meta,
      porDia: dados.porDia,
      totais: dados.totais
    };
  }

  return ContentService.createTextOutput(JSON.stringify(resultado))
    .setMimeType(ContentService.MimeType.JSON);
}

function processarDados(ss, nomeAba) {
  var sheet = ss.getSheetByName(nomeAba);
  var resultado = {
    porDia: {},
    totais: { enviados: 0, respostas: 0, respostasCS: 0 }
  };

  if (!sheet) return resultado;

  var lastRow = sheet.getLastRow();
  if (lastRow < 2) return resultado;

  // Colunas: A = Status Disparo, B = Mensagem Agente, C = Mensagem Aluno
  var range = sheet.getRange(2, 1, lastRow - 1, 3);
  var values = range.getValues();

  for (var i = 0; i < values.length; i++) {
    var statusRaw = String(values[i][0]).trim();
    var msgAgente = String(values[i][1]).trim();
    var msgAluno = String(values[i][2]).trim();

    if (statusRaw === "" || statusRaw === "undefined") continue;

    // Extrair cada check verde com sua data
    var partes = statusRaw.split("+");
    for (var p = 0; p < partes.length; p++) {
      var parte = partes[p].trim();
      if (parte.indexOf("✅") === -1) continue; // Ignora partes com X ou sem check

      // Extrair data do formato: "✅ DD/MM/YYYY HH:MM"
      var matchData = parte.match(/(\d{2})\/(\d{2})\/(\d{4})/);
      if (!matchData) continue;

      var dia = matchData[1];
      var mes = matchData[2];
      var ano = matchData[3];
      var chave = ano + "-" + mes + "-" + dia; // formato YYYY-MM-DD

      if (!resultado.porDia[chave]) {
        resultado.porDia[chave] = { enviados: 0, respostas: 0, respostasCS: 0 };
      }

      resultado.porDia[chave].enviados++;
      resultado.totais.enviados++;
    }

    // Mensagem Aluno: extrair data se existir
    if (msgAluno !== "" && msgAluno !== "undefined") {
      var matchAluno = msgAluno.match(/(\d{2})\/(\d{2})\/(\d{4})/);
      if (matchAluno) {
        var chaveAluno = matchAluno[3] + "-" + matchAluno[2] + "-" + matchAluno[1];
        if (!resultado.porDia[chaveAluno]) {
          resultado.porDia[chaveAluno] = { enviados: 0, respostas: 0, respostasCS: 0 };
        }
        resultado.porDia[chaveAluno].respostas++;
      }
      resultado.totais.respostas++;
    }

    // Mensagem Agente: extrair data se existir
    if (msgAgente !== "" && msgAgente !== "undefined") {
      var matchAgente = msgAgente.match(/(\d{2})\/(\d{2})\/(\d{4})/);
      if (matchAgente) {
        var chaveAgente = matchAgente[3] + "-" + matchAgente[2] + "-" + matchAgente[1];
        if (!resultado.porDia[chaveAgente]) {
          resultado.porDia[chaveAgente] = { enviados: 0, respostas: 0, respostasCS: 0 };
        }
        resultado.porDia[chaveAgente].respostasCS++;
      }
      resultado.totais.respostasCS++;
    }
  }

  return resultado;
}
