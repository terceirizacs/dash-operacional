function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  var clientesConfig = [
    { dadosAba: "Andrea Vermont", kpiAba: "Operação Semanal Andrea", meta: 1000 },
    { dadosAba: "Dra Investidora", kpiAba: "Operação Semanal Dra Investidor", meta: 50 },
    { dadosAba: "Rafa Bueno", kpiAba: "Operação Semanal Rafa Bueno", meta: 250 },
    { dadosAba: "Bruno Lucarelli", kpiAba: "Operação Semanal Bruno Lucarell", meta: 200 }
  ];

  var resultado = { clientes: {}, atualizadoEm: new Date().toLocaleString("pt-BR") };

  for (var i = 0; i < clientesConfig.length; i++) {
    var cfg = clientesConfig[i];
    var nome = cfg.dadosAba;

    // Ler aba de dados
    var dados = lerDados(ss, cfg.dadosAba);

    // Ler aba de KPIs
    var kpis = lerKPIs(ss, cfg.kpiAba);

    resultado.clientes[nome] = {
      meta: cfg.meta,
      kpis: kpis,
      dados: dados
    };
  }

  return ContentService.createTextOutput(JSON.stringify(resultado))
    .setMimeType(ContentService.MimeType.JSON);
}

function lerDados(ss, nomeAba) {
  var sheet = ss.getSheetByName(nomeAba);
  if (!sheet) return { total: 0, enviados: 0, erros: 0, comResposta: 0, semResposta: 0, comRespostaAgente: 0 };

  var lastRow = sheet.getLastRow();
  if (lastRow < 2) return { total: 0, enviados: 0, erros: 0, comResposta: 0, semResposta: 0, comRespostaAgente: 0 };

  var range = sheet.getRange(2, 1, lastRow - 1, 3);
  var values = range.getValues();

  var total = 0, enviados = 0, erros = 0, comResposta = 0, semResposta = 0, comRespostaAgente = 0;

  for (var i = 0; i < values.length; i++) {
    var status = String(values[i][0]).trim();
    var msgAgente = String(values[i][1]).trim();
    var msgAluno = String(values[i][2]).trim();

    if (status === "" || status === "undefined") continue;

    total++;

    if (status.indexOf("✅") !== -1) {
      enviados++;
    }
    if (status.indexOf("❌") !== -1) {
      erros++;
    }

    if (msgAluno !== "" && msgAluno !== "undefined") {
      comResposta++;
    } else {
      semResposta++;
    }

    if (msgAgente !== "" && msgAgente !== "undefined") {
      comRespostaAgente++;
    }
  }

  return {
    total: total,
    enviados: enviados,
    erros: erros,
    comResposta: comResposta,
    semResposta: semResposta,
    comRespostaAgente: comRespostaAgente
  };
}

function lerKPIs(ss, nomeAba) {
  var sheet = ss.getSheetByName(nomeAba);
  if (!sheet) return {};

  var valuesRow = sheet.getRange(6, 1, 1, 31).getValues()[0];

  function getVal(colIndex) {
    var v = valuesRow[colIndex - 1];
    if (v === "" || v === null || v === undefined) return 0;
    if (typeof v === "string") {
      v = v.replace("%", "").replace(",", ".").trim();
      var num = parseFloat(v);
      return isNaN(num) ? 0 : num;
    }
    return v;
  }

  return {
    enviadas: getVal(3),
    respostas: getVal(5),
    taxaResposta: getVal(7),
    enviadasHoje: getVal(9),
    respostasHoje: getVal(11),
    respostasCSHoje: getVal(15),
    respostasAlunosHoje: getVal(17),
    respostasCSOnttem: getVal(19),
    respostasAlunosOntem: getVal(21),
    respostaCSSemana: getVal(23),
    respostaAlunoSemana: getVal(25),
    pendente: getVal(27),
    meta: getVal(29),
    alcanceMeta: getVal(31)
  };
}
